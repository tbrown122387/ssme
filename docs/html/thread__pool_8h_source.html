232  std::lock_guard<std::mutex> out_lock{m_output_mut};
233  func_output_t m = *max_element(m_all_calcs.begin(), m_all_calcs.end());
234  func_output_t sum_exp = 0;
235  for(unsigned i = 0; i < m_total_calcs; ++i){
236  sum_exp += std::exp(m_all_calcs[i] - m);
237  }
238  func_output_t final_val = m + std::log(sum_exp) - std::log(m_total_calcs);
239 
240  if constexpr(debug)

242  m_out.set_value(final_val);
243 
244  // keep going until work is exhausted
245  m_count++;
246 
247  } else if (calc_count < m_total_calcs) {
248 
249  // call the expensive function that returns an approximate log-likelihood
250  func_output_t val;
251  {
252  std::shared_lock<std::shared_mutex> param_lock(m_input_mut);
253  val = m_f(m_param, m_observed_data);
254  }
255

262  }
