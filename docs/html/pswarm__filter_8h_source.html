97  [[maybe_unused]] unsigned num_terms_in_thread){
98 
99  // set up required variables

103  res.second += vec_mats_and_like.second / static_cast<float_t>(num_threads);
104 
105  // check if agg has at least one term in it

107  bool agg_old = false;

110  agg_old = true;

112 
113  // aggregate expectation matrices
114  // if agg has just been initialized, it is a vector of 0X0 matrices
115  // you can't add matrices to each of these 0X0 matrices at the first time point
116  if( agg_old ){
117  for(size_t i = 0; i < n_filt_funcs; ++i)
118  res.first[i] = res.first[i] + vec_mats_and_like.first[i]/ static_cast<float_t>(num_threads);
119  }else{
120  for(size_t i = 0; i < n_filt_funcs; ++i)
121  res.first[i] = vec_mats_and_like.first[i] / static_cast<float_t>(num_threads);
122  }
