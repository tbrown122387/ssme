91 void mn_resamp_states_and_params<nparts,dimx,dimparam,float_t>::resampLogWts(arrayVec &oldStateParts, arrayParams &oldParamParts, arrayFloat &oldLogUnNormWts)
92 {
93  // Using the fancier algorthm detailed on page 244 of IHMM
94 
95  // Create unnormalized weights
96  arrayFloat unnorm_weights;
97  float_t m = *std::max_element(oldLogUnNormWts.begin(), oldLogUnNormWts.end());
98  std::transform(oldLogUnNormWts.begin(), oldLogUnNormWts.end(), unnorm_weights.begin(),
99  [&m](float_t& d) -> float_t { return std::exp( d - m ); } );
100 
101  // two things:
102  // 1.) calculate normalizing constant for weights, and
103  // 2.) generate all these exponentials to help with getting order statistics
104  // NB: you never need to store E_{N+1}! (this is subtle)
105  float_t weight_norm_const(0.0);
106  arrayFloat exponentials;
107  float_t G(0.0);
108  for(size_t i = 0; i < nparts; ++i) {
109  weight_norm_const += unnorm_weights[i];
110  exponentials[i] = -std::log(m_u_sampler(this->m_gen));
111  G += exponentials[i];
112  }
113  G -= std::log(m_u_sampler(this->m_gen)); // E_{N+1}
114 
115  // see Fig 7.15 in IHMM on page 243
116  arrayVec tmpStatePartics = oldStateParts;
117  arrayParams tmpParamPartics = oldParamParts;
118  float_t uniform_order_stat(0.0); // U_{(i)} in the notation of IHMM
119  float_t running_sum_normalized_weights(unnorm_weights[0]/weight_norm_const); // \sum_{j=1}^I \omega^j in the notation of IHMM
120  float_t one_less_summand(0.0); // \sum_{j=1}^{I-1} \omega^j
121  unsigned int idx = 0;
122  for(size_t i = 0; i < nparts; ++i){
123 
124  uniform_order_stat += exponentials[i]/G; // add a spacing E_i/G
125 
126  do {
127  if( (one_less_summand < uniform_order_stat) && (uniform_order_stat <= running_sum_normalized_weights) ) {
128  // select index idx
129  tmpStatePartics[i] = oldStateParts[idx];
130  tmpParamPartics[i] = oldParamParts[idx];
131  break;
132  }else{
133  // increment idx because it will never be chosen (all the other order statistics are even higher)
134  idx++;
135  running_sum_normalized_weights += unnorm_weights[idx]/weight_norm_const;
136  one_less_summand += unnorm_weights[idx-1]/weight_norm_const;
137  }
138  }while(true);
139  }
140 
141  //overwrite olds with news
142  oldStateParts = std::move(tmpStatePartics);
143  oldParamParts = tmpParamPartics;
144  std::fill(oldLogUnNormWts.begin(), oldLogUnNormWts.end(), 0.0);
145 }

999  m2 = logFirstStageUnNormWeights[ii];

1011  arrayUInt myKs = m_kGen.sample(logFirstStageUnNormWeights);
1012

1026  second_cll_sum += std::exp( logFirstStageUnNormWeights[ii] - m2 );
1027  third_cll_sum += std::exp( m_logUnNormWeights[ii] - m3 );
1028

1030  xtm1k = old_state_partics[myKs[ii]];
1031  mtm1k = m_a * old_param_partics[myKs[ii]].get_trans_params() + (1.0 - m_a) * m_thetaBar;
1032  param::pack<float_t, dimparam> mtm1k_pack (mtm1k, m_transforms);
1033  m_mvn_gen.setMean(mtm1k);
1034  param::pack<float_t, dimparam> newThetaSamp (m_mvn_gen.sample(), m_transforms);
1035

1037  m_param_particles[ii] = newThetaSamp;
1038  m_state_particles[ii] = fSamp(xtm1k, cov_data, newThetaSamp.get_untrans_params());
1039  muTk = propMu(xtm1k, cov_data, old_param_partics[myKs[ii]].get_untrans_params());
1040  m_logUnNormWeights[ii] = logGEv(obs_data, m_state_particles[ii], newThetaSamp.get_untrans_params())
1041  - logGEv(obs_data, muTk, mtm1k_pack.get_untrans_params());
1042 
1043  if constexpr(debug) {
1044  std::cout << "time: " << m_now

1047  << ", log unnorm weight: " << m_logUnNormWeights[ii]
1048  << "\n";
1049  }
1050

1052  if(m_logUnNormWeights[ii] > m1)
1053  m1 = m_logUnNormWeights[ii];
1054  }
1055 
1056  // calculate estimate for log of last conditonal likelihood
1057  for(size_t p = 0; p < nparts; ++p)
1058  first_cll_sum += std::exp( m_logUnNormWeights[p] - m1 );
1059  m_logLastCondLike = m1 + std::log(first_cll_sum) + m2 + std::log(second_cll_sum) - 2*m3 - 2*std::log(third_cll_sum);
1060 
1061  if constexpr(debug)
1062  std::cout << "time: " << m_now << ", log cond like: " << m_logLastCondLike << "\n";
1063 
1064  // calculate expectations before you resample
1065  unsigned int fId(0);
1066  for(auto & h : fs){
1067 
1068  Mat testOutput = h(m_state_particles[0], cov_data, m_param_particles[0].get_untrans_params()); //TODO
1069  unsigned int rows = testOutput.rows();
1070  unsigned int cols = testOutput.cols();
1071  Mat numer = Mat::Zero(rows,cols);
1072  float_t denom(0.0);
1073 
1074  for(size_t prtcl = 0; prtcl < nparts; ++prtcl){
1075  numer += h(m_state_particles[prtcl], cov_data, m_param_particles[prtcl].get_untrans_params()) * std::exp(m_logUnNormWeights[prtcl] - m1);//TODO
1076  denom += std::exp(m_logUnNormWeights[prtcl] - m1);
1077  }

1079 
1080  if constexpr(debug)
1081  std::cout << "transposed expectation " << fId << "; " << m_expectations[fId] << "\n";
1082 
1083  fId++;
1084  }
1085 
1086  // if you have to resample
1087  if( (m_now+1)%m_rs == 0)
1088  m_resampler.resampLogWts(m_state_particles, m_param_particles, m_logUnNormWeights);
1089 
1090  // advance time
1091  m_now += 1;
1092 
1093  } else { // (m_now == 0)
1094 
1095  float_t max(-std::numeric_limits<float_t>::infinity());
1096  psv untrans_param_samp;
1097  for(size_t ii = 0; ii < nparts; ++ii)
1098  {
1099  // sample param particles
1100  untrans_param_samp = paramPriorSamp();
1101  m_param_particles[ii] = param::pack<float_t,dimparam>(untrans_param_samp, m_transforms, false);
1102 
1103  // sample particles
1104  m_state_particles[ii] = q1Samp(obs_data, untrans_param_samp);

1119  max = m_logUnNormWeights[ii];
1120  }

1125  sumExp += std::exp( m_logUnNormWeights[i] - max );
1126  }

1140  numer += h(m_state_particles[prtcl], cov_data, m_param_particles[prtcl].get_untrans_params()) * std::exp(m_logUnNormWeights[prtcl] - max);//TODO
1141  denom += std::exp(m_logUnNormWeights[prtcl] - max);
