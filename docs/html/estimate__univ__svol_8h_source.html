60 double univ_svol_estimator<numparams,dimnss,dimss,dimobs,numparts>::logPriorEvaluate(const paramPack& theta)
61 {
62  double returnThis(0.0);
63 
64  // type aliases
65  using sssv = Eigen::Matrix<double,dimss,1>;
66 
67  // 1 phi, 1 mu, 1 ss
68  // unpack parameters
69  double phi, mu, ss;
70  unpackUnivSvolParams<dimss,dimobs>(theta, beta, phis, mus, sigmas, RSigmasVec, lambda, p);
71 
72  // betas
73  for(size_t i = 0; i < dimobs-1; ++i)
74  returnThis += rveval::evalUnivNorm(beta(i+1), 1.0, std::sqrt(.125), true);
75 
76  // phis
77  for(size_t i = 0; i < dimss; ++i)
78  returnThis += rveval::evalUniform(phis(i), .4, .9, true);
79 
80  // mus
81  for(size_t i = 0; i < dimss; ++i)
82  returnThis += rveval::evalUnivNorm(mus(i),0.0,1.0,true);
83 
84  // sigmas
85  for(size_t i = 0; i < dimss; ++i)
86  returnThis += rveval::evalUnivInvGamma(sigmas(i)*sigmas(i), 1.0, 1.0, true);
87 
88  // r variances now
89  for(size_t i = 0; i < dimobs; ++i)
90  returnThis += rveval::evalUnivInvGamma(RSigmasVec(i)*RSigmasVec(i), .001, .001, true);
91 
92  // lambda
93  returnThis += rveval::evalUniform(lambda, 1.5e-4, .002708178, true);
94 
95  // ps
96  returnThis += rveval::evalUniform(p, 0.5, 1.0, true);
97 
98 
99  return returnThis;
100 
101 }
