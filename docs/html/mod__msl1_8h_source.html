13 template<size_t dimss, size_t dimy>
14 void unpackMSL1Params(
15  const paramPack& packedParams,
16  Eigen::Matrix<double,dimy,1>& beta,
17  Eigen::Matrix<double,dimss,1>& phis,
18  Eigen::Matrix<double,dimss,1>& mus,
19  Eigen::Matrix<double,dimss,1>& sigmas,
20  Eigen::Matrix<double,dimy,1>& RSigmasVec,
21  double& lambda,
22  double& p)
23 {
24  // assumes dimss==2
25  if(packedParams.getNumParams() != (dimy-1)+(3*dimss) + (dimy) + 2)
26  throw std::runtime_error("the parameter dimensions don't line up\n");
27 
28  // assumed order: beta, phis, mus, sigmas, RSigmas, lambasPosAndNeg, p
29  // for more information on this logic, see the hand-written notesheet
30  beta = Eigen::Matrix<double,dimy,1>::Zero();
31  // assuming that dim_factor is 1!
32  beta(0,0) = 1.0;
33  // recall that htere are dimy*dimx - dimx - (dimx C 2) elements in the beta matrix
34  beta.block(1, 0, dimy-1,1) = packedParams.getUnTransParams(0, dimy-2);
35  unsigned int G = dimy-1;
36  phis = packedParams.getUnTransParams(G, G+dimss-1);
37  mus = packedParams.getUnTransParams(G+dimss, G+2*dimss-1);
38  sigmas = packedParams.getUnTransParams(G+2*dimss,G+3*dimss-1).array().sqrt().matrix();
39  RSigmasVec = packedParams.getUnTransParams(G+3*dimss, G+3*dimss+dimy-1).array().sqrt().matrix();
40  lambda = packedParams.getUnTransParams(G+3*dimss+dimy, G+3*dimss+dimy)(0);
41  p = packedParams.getUnTransParams(G+4*dimss+dimy-1,G+4*dimss+dimy-1)(0);
42 }

156 template<size_t nparts, size_t dimnss, size_t dimss, size_t dimy, typename resampT>
157 msl1_rbbpf<nparts,dimnss,dimss,dimy,resampT>::msl1_rbbpf(const unsigned int &K, const paramPack& pp, bool calc_d_aves)
158  : rbpf_hmm_bs<nparts, dimnss, dimss, dimy, resampT>() // always resamples everytime
159  , m_K(K), m_store_dbar_aves(calc_d_aves)
160 {
161  // unpack parameters
162  unpackMSL1Params<dimss,dimy>(pp, m_beta_1, m_phis, m_mus, m_sigmas, m_R_sigmas, m_lambda, m_p);
163 
164  // set up time one sampler guy
165  m_timeOneSampler.setMean(sssv::Zero());
166  sssMat sigma0 = ( m_sigmas.array().square() / (1.0 - m_phis.array().square()) ).matrix().asDiagonal();
167  m_timeOneSampler.setCovar(sigma0);
168 
169  // and set up trans jump sampler
170  m_transJumpSampler.setMean(sssv::Zero());
171  m_transJumpSampler.setCovar(m_sigmas.array().square().matrix().asDiagonal());
172 
173  // store the entire lexicographic ordering of the 0,1,2..K hot vectors
174  unsigned int ctr = 0;
175  for(size_t hotNumber = 0; hotNumber < m_K+1; ++hotNumber){
176 
177  // changes each iteration...std::vector because it plays nicely with std::next_permutation
178  std::vector<double> dynVec(dimy, 0.0);
179 
180  // fill in the ones you need and sort
181  for(size_t oneIdx = 0; oneIdx < hotNumber; ++oneIdx){ // (0,0,0), (0,0,1), (0,1,1)
182  dynVec[oneIdx] = 1.0;
183  }
184  std::sort(dynVec.begin(), dynVec.end());
185 
186  //add permutations to allVecs
187  do {
188  double* ptr = &dynVec[0];
189  Eigen::Map<osv> tmp(ptr, dimy); // TODO: do we need dimy?
190  m_all_indic_vecs[ctr] = tmp;
191  ctr++;
192  } while(std::next_permutation(dynVec.begin(), dynVec.end()));
193  }
194 
195  // store dbar and dbbd_bar
196  if(m_store_dbar_aves){
197 
198  m_dbbdbar = osm::Zero();
199  osm D;
200  for(size_t x1t = 0; x1t < dimnss; ++x1t){ // for each value of x1t
201 
202  // we can use the marginal distribution of x_{1,t}'s weights
203  // for more info see appendix
204  D = m_all_indic_vecs[x1t].asDiagonal();
205  m_dbbdbar += D * m_beta_1 * m_beta_1.transpose() * D / (double)dimnss;
206  }
207  }
208 }

219 template<size_t nparts, size_t dimnss, size_t dimss, size_t dimy, typename resampT>
220 auto msl1_rbbpf<nparts,dimnss,dimss,dimy,resampT>::get_conditional_var(const sssv& x2t) const -> osm
221 {
222  // for a derivation of this expression, see the appendix of the msvol paper
223  if(!m_store_dbar_aves) throw std::runtime_error("you need to be storing dbars to calculate these conditional covariances\n");
224  osm R = m_R_sigmas.cwiseProduct(m_R_sigmas).asDiagonal();
225  return std::exp(m_mus(0) + m_phis(0)*(x2t(0) - m_mus(0)) + .5*m_sigmas(0)*m_sigmas(0))*m_beta_1*m_beta_1.transpose() +
226  std::exp(m_mus(1) + m_phis(1)*(x2t(1) - m_mus(1)) + .5*m_sigmas(1)*m_sigmas(1))*m_dbbdbar + R;
227 }
