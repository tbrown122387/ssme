160 template<size_t numparams, size_t dimobs, size_t numparts>
161 void ada_pmmh_mvn<numparams,dimobs,numparts>::update_moments_and_Ct(const paramPack& newTheta)
162 {
163  // if m_iter = 1, that means we're on iteration 2,
164  // but we're calling this based on the previous iteration,
165  // so that's iteration 1, think of n as the actual
166  // number of iterations that have happened then (not counting from 0)
167  // On the other hand, yu might not want to count the first iteration,
168  // though, because we didn't "propose" anything and probabilistically accept/reject it
169  psv newTransTheta = newTheta.getTransParams();
170  if(m_iter == 1){
171  // at the moment m_meanTransTheta is zero, so we add m_currentTransTheta
172  // we'll also leave m_sigma_hat to be zero for the time being
173  m_mean_trans_theta += newTransTheta;
174  }else if(m_iter == 2){
175  // at the beginning of this call, m_mean_trans_theta is x1*x1^T and m_sigma_hat is zero
176  // this is a an algebraically simplified version of the split up non-recursive formula
177  m_sigma_hat = m_mean_trans_theta*m_mean_trans_theta.transpose()
178  + newTransTheta * newTransTheta.transpose()
179  - m_mean_trans_theta * newTransTheta.transpose()
180  - newTransTheta * m_mean_trans_theta.transpose();
181  m_sigma_hat *= .5;
182  m_mean_trans_theta = .5* m_mean_trans_theta + .5*newTransTheta;
183  }else if( m_iter > 2){
184  // regular formula. see
185  // https://stats.stackexchange.com/questions/310680/sequential-recursive-online-calculation-of-sample-covariance-matrix/310701#310701
186  m_sigma_hat = m_sigma_hat*(m_iter-2.0)/(m_iter-1.0)
187  + (newTransTheta - m_mean_trans_theta)*(newTransTheta - m_mean_trans_theta).transpose()/m_iter;
188  m_mean_trans_theta = ((m_iter-1.0)*m_mean_trans_theta + newTransTheta)/m_iter;
189 
190  }else{
191  std::cerr << "something went wrong\n";
192  }
193 
194  // now update Ct
195  if( (m_t1 > m_iter) && (m_iter > m_t0) ) // in window, so we want to adjust Ct
196  m_Ct = m_sd * ( m_sigma_hat + m_eps * psm::Identity() );
197 
198 }
