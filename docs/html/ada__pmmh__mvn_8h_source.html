211 template<size_t numparams, size_t dimobs, size_t numparts, typename float_t>
212 void ada_pmmh_mvn<numparams,dimobs,numparts,float_t>::update_moments_and_Ct(const param::pack<float_t,numparams>& new_theta)
213 {
214  // if m_iter = 1, that means we're on iteration 2,
215  // but we're calling this based on the previous iteration,
216  // so that's iteration 1, think of n as the actual
217  // number of iterations that have happened then (not counting from 0)
218  // On the other hand, yu might not want to count the first iteration,
219  // though, because we didn't "propose" anything and probabilistically accept/reject it
220  psv new_trans_theta = new_theta.get_trans_params();
221  if(m_iter == 1){
222  // at the moment m_meanTransTheta is zero, so we add m_currentTransTheta
223  // we'll also leave m_sigma_hat to be zero for the time being
224  m_mean_trans_theta += new_trans_theta;
225  }else if(m_iter == 2){
226  // at the beginning of this call, m_mean_trans_theta is x1*x1^T and m_sigma_hat is zero
227  // this is a an algebraically simplified version of the split up non-recursive formula
228  m_sigma_hat = m_mean_trans_theta*m_mean_trans_theta.transpose()
229  + new_trans_theta * new_trans_theta.transpose()
230  - m_mean_trans_theta * new_trans_theta.transpose()
231  - new_trans_theta * m_mean_trans_theta.transpose();
232  m_sigma_hat *= .5;
233  m_mean_trans_theta = .5* m_mean_trans_theta + .5*new_trans_theta;
234  }else if( m_iter > 2){
235  // regular formula. see
236  // https://stats.stackexchange.com/questions/310680/sequential-recursive-online-calculation-of-sample-covariance-matrix/310701#310701
237  m_sigma_hat = m_sigma_hat*(m_iter-2.0)/(m_iter-1.0)
238  + (new_trans_theta - m_mean_trans_theta)*(new_trans_theta - m_mean_trans_theta).transpose()/m_iter;
239  m_mean_trans_theta = ((m_iter-1.0)*m_mean_trans_theta + new_trans_theta)/m_iter;
240 
241  }else{
242  std::cerr << "something went wrong\n";
243  }
244 
245  // now update Ct
246  if( (m_t1 > m_iter) && (m_iter > m_t0) ) // in window, so we want to adjust Ct
247  m_Ct = m_sd * ( m_sigma_hat + m_eps * psm::Identity() );
248 
249 }

324 template<size_t numparams, size_t dimobs, size_t numparts, typename float_t>
325 void ada_pmmh_mvn<numparams,dimobs,numparts,float_t>::commence_sampling()
326 {
327 
328  // random number stuff to decide on whether to accept or reject
329  rvsamp::UniformSampler<float_t> runif;
330  psv proposed_trans_theta;
331  while(m_iter < m_num_mcmc_iters) // every iteration
332  {
333 
334  if(m_iter > 0) {
335 
336  // update sample moments (with the parameters that were just accepted) and Ct so we can q_samp()
337  update_moments_and_Ct(m_current_theta);
338 
339  // propose a new theta
340  proposed_trans_theta = q_samp(m_current_theta);
341  dyn_data_t proposed_theta(proposed_trans_theta, m_tts);
342  m_new_log_prior = log_prior_eval(proposed_theta) + proposed_theta.get_log_jacobian();
343  m_new_log_like = m_pool.work(proposed_theta);
344 
345  // decide whether to accept or reject
346  m_log_accept_prob = m_new_log_prior + m_new_log_like - m_old_log_prior - m_old_log_like;
347  float_t log_uniform_draw = std::log(runif.sample());
348  m_accepted = log_uniform_draw < m_log_accept_prob; // TODO: verify that everything compared to a NaN is false!
349  if(m_accepted){
350  m_ma_accept_rate = 1.0/(m_iter+1.0) + m_iter*m_ma_accept_rate/(m_iter+1.0);
351  m_current_theta = proposed_theta;
352  m_old_log_prior = m_new_log_prior;
353  m_old_log_like = m_new_log_like;
354  }else{
355  m_ma_accept_rate = 0.0/(m_iter+1.0) + m_iter*m_ma_accept_rate/(m_iter+1.0);
356  if( std::isnan(m_log_accept_prob) ) {
357  std::cerr << "accept proability had a nan in it\n";
358  }
359  }
360 
361  }else{ // first iteration
362  m_old_log_like = m_pool.work(m_current_theta);
363  m_old_log_prior = log_prior_eval(m_current_theta) + m_current_theta.get_log_jacobian();
364  }
365 
366  record_params();
367  record_messages();
368  m_iter++;
369 
370  }
371 }
